% Bifrost technical documentation — Hydrozoa-inspired LaTeX preamble
% Included into pandoc via --include-in-header. Sits after hyperref is loaded
% by pandoc's default template, so \hypersetup overrides work.

% -------------------------------------------------------------------
% Colors
% -------------------------------------------------------------------
\definecolor{link-blue}{HTML}{074DB5}
\definecolor{inline-code-dark-red}{HTML}{9A2626}
\definecolor{cover-page-blue}{HTML}{03314D}
\definecolor{shadecolor}{HTML}{EEEEEE}

% -------------------------------------------------------------------
% Typography
% -------------------------------------------------------------------
\linespread{1.08}
\setlength{\parskip}{0.8em plus 0.3em minus 0.2em}
\setlength{\parindent}{0pt}
\usepackage{microtype}
\hfuzz=0.8em  % relax overfull \hbox warnings

% -------------------------------------------------------------------
% Hyperlinks — override pandoc's defaults with a single blue.
% Pandoc loads hyperref *after* header-includes, so defer until then.
% -------------------------------------------------------------------
\AtBeginDocument{%
  \hypersetup{%
    colorlinks = true,%
    linkcolor  = link-blue,%
    citecolor  = link-blue,%
    urlcolor   = link-blue,%
    breaklinks = true%
  }%
}

% -------------------------------------------------------------------
% Inline code — red monospace (Hydrozoa signature look).
% Pandoc renders inline code as \texttt{...}; redefine it to colorize.
% -------------------------------------------------------------------
\DeclareRobustCommand{\texttt}[1]{{\color{inline-code-dark-red}\ttfamily #1}}

% -------------------------------------------------------------------
% Lists — Hydrozoa-style generous left margin
% -------------------------------------------------------------------
\usepackage{enumitem}
\setlist[itemize]{leftmargin=2.4em, labelindent=0.6em}
\setlist[enumerate]{leftmargin=2.4em, labelindent=0.6em}
\setlist[description]{leftmargin=2.4em, labelindent=0.6em}

% -------------------------------------------------------------------
% Tables — crisp rules
% -------------------------------------------------------------------
\usepackage{booktabs}

% -------------------------------------------------------------------
% Figures — thin box around every figure float
% -------------------------------------------------------------------
\usepackage{float}
\floatstyle{boxed}
\restylefloat{figure}

% -------------------------------------------------------------------
% Page footer — page number centered, git watermark on the right
% Relies on \gitRev / \gitDate / \gitBranch provided by the build script
% via a companion --include-in-header file.
% -------------------------------------------------------------------
\providecommand{\gitRev}{}
\providecommand{\gitDate}{}
\providecommand{\gitBranch}{}

\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\fancyfoot[C]{\thepage}
\fancyfoot[R]{%
  \footnotesize\color{gray}%
  Branch:~\textbf{\gitBranch}~@~\gitRev~\textbullet{}~\gitDate%
}
